    print("Accuracy: ", acc)
    print("Precision:", pre)

    print("F1-Score: ", f1)
    if isinstance(soft_label, np.ndarray):
        print("Auc Score:", auc)
    print("-" * 31)

def x_learner_predict(data, score_model, model_step2):
    score = score_model.predict_proba(data)
    y_list = []

    for i in range(treat_class_num - 1):
        r_0 = model_step2[i][0].predict(data)
        r_i = model_step2[i][1].predict(data)
        e_0 = score[:, 0] / (score[:,0] + score[:,i+1])
        e_i = score[:, i+1] / (score[:,0] + score[:,i+1])
        y_i = e_0 * r_0 + e_i * r_i
        y_i = np.expand_dims(y_i, axis=1)
        y_list.append(y_i)
    y_list = np.concatenate(y_list, axis = 1)

    cate = np.max(y_list, axis=1)
    best_t = np.argmax(y_list, axis=1)
    best_t += 1
    
    idx = np.where(cate < 0)[0]
    best_t[idx] = 0
    
    return cate, best_t
